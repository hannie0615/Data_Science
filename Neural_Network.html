class CoordinateDataset(Dataset):   
    def __init__(self):
        #Enter the address of data
        csvData=np.loadtxt('C:/Users/sksms/Desktop/BioStatics/data_spiral.csv',delimiter=',',dtype=np.float32)
        self.len=csvData.shape[0]
        self.X=torch.from_numpy(csvData[:,:2]) #coordinates
        self.Y=torch.from_numpy(csvData[:,-1]).type(torch.LongTensor) #class
        
    def __getitem__(self,index):
        return self.X[index], self.Y[index]
    
    def __len__(self):
        return self.len

dataset = CoordinateDataset()
train_loader = DataLoader(dataset = dataset,
                         batch_size=32, #  
                         shuffle=True,
                         num_workers=0)

class Net(torch.nn.Module):
    
    def __init__(self):
        super(Net,self).__init__()
        self.relu = torch.nn.ReLU()
        self.tanh = torch.nn.Tanh()
        self.l1=torch.nn.Linear(2,16)
        self.l2=torch.nn.Linear(16,8)
        self.l3=torch.nn.Linear(8,2)        
        
        
    def forward(self,x):
        x = self.tanh(self.l1(x)) #  tanh, relu
        x = self.relu(self.l2(x))

        return x

model = Net()

criterion = torch.nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(),lr=0.01)
X1=np.zeros(200)
X2=np.zeros(200)

# Training loop
print("START EPOCH")
for epoch in range(200): # epoch  iteration: 200 
    if epoch%50==0:
        print("EPOCH : {} / 200".format(epoch))
    for i, data in enumerate(train_loader):
        inputs, labels = data
        inputs, labels = Variable(inputs), Variable(labels)

        y_pred = model(inputs)

        loss = criterion(y_pred, labels)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        X1[epoch]=epoch
        X2[epoch]=loss
        
        
print("END")
